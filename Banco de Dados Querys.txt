CREATE TABLE usuarios(
    idUsuario INT GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY,
    loginUsuario VARCHAR(50),
    senhaUsuario VARCHAR(50),
    emailUsuario VARCHAR(100)
);

CREATE TABLE produtos(
idProduto INT GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1),
nomeProduto VARCHAR(100),
valorProduto NUMERIC(11,2),
imgProduto VARCHAR(100),
CONSTRAINT produtopk PRIMARY KEY (idProduto, valorProduto)
);

CREATE table carrinho(
    idProduto INT,
    idUsuario INT,
    quantProduto NUMERIC,
    valorProduto NUMERIC (11,2),
    CONSTRAINT produtofk FOREIGN KEY (idProduto, valorProduto) REFERENCES produtos (idProduto, valorProduto),
    CONSTRAINT usuariofk FOREIGN KEY (idUsuario) REFERENCES usuarios (idUsuario)
);
